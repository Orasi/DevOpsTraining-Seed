POST /users
Create a new user

Only accessible by Admins

Examples

POST /users/
{
  "user": {
    "first_name": "Walton",
    "last_name": "Kris",
    "password": "1234",
    "password_confirmation": "1234",
    "username": "neil",
    "email": "rebekah@schaden.name"
  }
}
200
{
  "user": {
    "id": 5684,
    "username": "neil",
    "email": "rebekah@schaden.name",
    "first_name": "Walton",
    "last_name": "Kris",
    "company": null,
    "admin": null,
    "created_at": "2017-05-17T16:26:37.977Z",
    "updated_at": "2017-05-17T16:26:37.977Z"
  }
}
