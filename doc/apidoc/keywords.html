Only accessible if current user can view parent project. Pass optional array of testcase ids to associate keyword to tests

{
  "keyword": {
    "keyword": "AWESOME",
    "project_id": 2104
  },
  "testcases": [
    10526,
    10527,
    10528,
    10529,
    10530

    "project_id": 2104,
    "testcase_count": 5
  }
}
